Angular

List Sorting and pagination
 Required Pagination template 
                            
$scope.currentPageRecords = 1;
$scope.pageSizeRecords = 25;

$scope.sortConsProducts = {
  column: '',
  descending: false
  };    
  
  $scope.changeSortingConsProducts = function(column) {
  
  var sortConsProducts = $scope.sortConsProducts;
  
  if (sortConsProducts.column == column) {
    sortConsProducts.descending = !sortConsProducts.descending;
  } else {
    sortConsProducts.column = column;
    sortConsProducts.descending = false;
  }
  };

Pagination template (must be attached as separate file)
 
                            
<ul class="pagination" ng-if="1 < pages.length">
<li class="page-item" ng-if="boundaryLinks" ng-class="{ disabled : pagination.current == 1 }">
  <a class="page-link" href="#" aria-label="Previous" ng-click="setCurrent(1)">
    <span aria-hidden="true">&laquo;</span>
  </a>
</li>
<li class="page-item" ng-if="directionLinks" ng-class="{ disabled : pagination.current == 1 }">
    <a class="page-link" href="#" aria-label="Previous" ng-click="setCurrent(pagination.current - 1)">
      <span aria-hidden="true">‹</span>
    </a>
  </li>
<li class="page-item" ng-repeat="pageNumber in pages track by $index" ng-class="{ active : pagination.current == pageNumber, disabled : pageNumber == '...' }">
    <a class="page-link" href="#" ng-click="setCurrent(pageNumber)">{{ pageNumber }}</a>
</li>
<li class="page-item" ng-if="directionLinks" ng-class="{ disabled : pagination.current == pagination.last }">
    <a class="page-link" href="#" aria-label="Next" ng-click="setCurrent(pagination.current + 1)">
      <span aria-hidden="true">›</span>
    </a>
</li>
<li class="page-item" ng-if="boundaryLinks"  ng-class="{ disabled : pagination.current == pagination.last }">
  <a class="page-link" href="#" aria-label="Next" ng-click="setCurrent(pagination.last)">
    <span aria-hidden="true">&raquo;</span>
  </a>
</li>
</ul>
